import pygame import random # Khởi tạo pygame pygame.init() # Cấu hình màn hình WIDTH, HEIGHT = 400, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Flappy Bird") # Màu sắc WHITE = (255, 255, 255) BLUE = (135, 206, 250) # Tải hình ảnh bird_img = pygame.image.load("bird.png") bird_img = pygame.transform.scale(bird_img, (50, 35)) # Cấu hình trò chơi gravity = 0.5 bird_movement = 0 bird_x, bird_y = 100, HEIGHT // 2 pipe_width = 70 pipe_gap = 150 pipe_x = WIDTH pipe_height = random.randint(100, 400) pipe_speed = 3 clock = pygame.time.Clock() running = True while running: screen.fill(BLUE) for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: bird_movement = -8 # Tạo lực hấp dẫn bird_movement += gravity bird_y += bird_movement # Vẽ chim screen.blit(bird_img, (bird_x, bird_y)) # Vẽ ống pygame.draw.rect(screen, WHITE, (pipe_x, 0, pipe_width, pipe_height)) pygame.draw.rect(screen, WHITE, (pipe_x, pipe_height + pipe_gap, pipe_width, HEIGHT - pipe_height - pipe_gap)) pipe_x -= pipe_speed # Kiểm tra va chạm if bird_y >= HEIGHT or bird_y <= 0 or ( pipe_x <= bird_x + 50 <= pipe_x + pipe_width and (bird_y <= pipe_height or bird_y + 35 >= pipe_height + pipe_gap)): running = False # Tạo ống mới if pipe_x + pipe_width < 0: pipe_x = WIDTH pipe_height = random.randint(100, 400) pygame.display.update() clock.tick(30) pygame.quit()
